\dayentry{2026-01-01}{Inequalities}{Green}
\begin{lemma}
For all real numbers $a,b$, we have $(a+b)^2 \le 2a^2 + 2b^2$.
\end{lemma}
\begin{proof}
Expand and use $2ab \le a^2 + b^2$.
\end{proof}

\dayentry{2026-01-02}{Linear Algebra}{Yellow}
\begin{proposition}
If $A$ is positive definite, then $x^\top A x > 0$ for all $x \ne 0$.
\end{proposition}
\begin{proof}
By definition of positive definiteness.
\end{proof}
